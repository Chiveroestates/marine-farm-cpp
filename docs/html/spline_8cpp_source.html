101 void Spline::compute_parameters()
102 {
103  a_.resize(n_, vector<Vector3f>(4));
104 
105  for (int k = 0; k < n_; k++) {
106  a_[k][0] = p_[k];
107  a_[k][1] = o_[k];
108  a_[k][2] = 3*(p_[k+1] - p_[k]) - 2*o_[k] - o_[k+1];
109  a_[k][3] = 2*(p_[k] - p_[k+1]) + o_[k] + o_[k+1];
110  }
